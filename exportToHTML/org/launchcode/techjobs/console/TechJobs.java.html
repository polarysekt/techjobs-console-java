74       // ﻿Returns the key of the selected item from the choices Dictionary 
75       private static String getUserSelection(String menuHeader, HashMap<String, String> choices) { 
76    
77           Integer choiceIdx; 
78           Boolean validChoice = false; 
79           String[] choiceKeys = new String[choices.size()]; 
80    
81           // Put the choices in an ordered structure so we can 
82           // associate an integer with each one 
83           Integer i = 0; 
84           for (String choiceKey : choices.keySet()) { 
85               choiceKeys[i] = choiceKey; 
86               i++; 
87           } 
88    
89           do { 
90    
91               System.out.println("\n" + menuHeader); 
92    
93               // Print available choices 
94               for (Integer j = 0; j < choiceKeys.length; j++) { 
95                   System.out.println("" + j + " - " + choices.get(choiceKeys[j])); 
96               } 
97    
98               choiceIdx = in.nextInt(); 
99               in.nextLine(); 
100   
101              // Validate user's input 
102              if (choiceIdx < 0 || choiceIdx >= choiceKeys.length) { 
103                  System.out.println("Invalid choice. Try again."); 
104              } else { 
105                  validChoice = true; 
106              } 
107   
108          } while(!validChoice); 
109   
110          return choiceKeys[choiceIdx]; 
111      }

113      // Print a list of jobs 
114      private static void printJobs(ArrayList<HashMap<String, String>> someJobs) { 
115   
116          if( someJobs.size() == 0 ) { 
117              System.out.println("*****"); 
118              System.out.println( "No data to display."); 
119          } else { 
120   
121              for (HashMap<String, String> i : someJobs) { 
122                  System.out.println("*****"); 
123                  for (HashMap.Entry<String, String> entry : i.entrySet()) { 
124                      System.out.printf("%s : %s\n", entry.getKey(), entry.getValue()); 
125                  } 
126              } 
127   
128          } 
129          System.out.println("*****"); 
130      }
